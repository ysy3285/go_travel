--고객정보
CREATE TABLE CUSTOM (
ID VARCHAR2(15) CONSTRAINT CUSTOM_ID_PK PRIMARY KEY,
PW VARCHAR2(15) NOT NULL,
NAME VARCHAR2(10) NOT NULL,
ADDR1 VARCHAR2(20),
ADDR2 VARCHAR2(30),
ADDR3 VARCHAR2(40),
TEL VARCHAR2(15) NOT NULL,
GRADE VARCHAR2(15));

--항공사 정보
CREATE TABLE AIR_COM (
AIRLINE VARCHAR2(20) CONSTRAINT AIR_COM_AIRLINE_PK PRIMARY KEY,
NATIONALITY VARCHAR2(20),
AIRLINE_TEL VARCHAR2(15));

--직원정보
CREATE TABLE EMPLOYEE (
EMPNO NUMBER(4) CONSTRAINT EMPLOYEE_EMPNO_PK PRIMARY KEY,
EMPNAME VARCHAR2(10) NOT NULL,
TEL VARCHAR2(15),
AIRLINE VARCHAR2(20) CONSTRAINT EMPLOYEE_AIRLINE_FK REFERENCES AIR_COM(AIRLINE),
POSITION VARCHAR2(10) NOT NULL);

--항공정보
CREATE TABLE AIRCRAFT (
AIRCRAFT_NUM VARCHAR2(7) CONSTRAINT AIRCRAFT_AIRCRAFT_NUM_PK PRIMARY KEY,
AIRLINE VARCHAR2(20) CONSTRAINT AIRCRAFT_AIRLINE_FK REFERENCES AIR_COM(AIRLINE),
MODEL VARCHAR2(20),
SEAT NUMBER(3),
PILOT NUMBER(4) CONSTRAINT AIRCRAFT_PILOT_FK REFERENCES EMPLOYEE(EMPNO),
CO_PILOT NUMBER(4) CONSTRAINT AIRCRAFT_CO_PILOT_FK REFERENCES EMPLOYEE(EMPNO),
CREW NUMBER(4) CONSTRAINT AIRCRAFT_CREW_FK REFERENCES EMPLOYEE(EMPNO));

--비행기 시간표
CREATE TABLE FLIGHT(
AIRLINE VARCHAR2(20) CONSTRAINT FLIGHT_AIRLINE_FK REFERENCES AIR_COM(AIRLINE),
AIRCRAFT_NUM VARCHAR2(7) CONSTRAINT FLIGHT_AIRCRAFT_NUM_FK REFERENCES AIRCRAFT(AIRCRAFT_NUM),
DEPART VARCHAR2(15),
ARRIVE VARCHAR2(15),
S_TIME VARCHAR2(10),
E_TIME VARCHAR2(10));

--예약
CREATE TABLE RESERVATION (
RENO VARCHAR2(4) CONSTRAINT RESERVATION_RENO_PK PRIMARY KEY,
ID VARCHAR2(15) CONSTRAINT RESERVATION_ID_FK REFERENCES CUSTOM(ID),
NAME VARCHAR2(10),
AIRLINE VARCHAR2(20) CONSTRAINT RESERVATION_AIRLINE_FK REFERENCES AIR_COM(AIRLINE),
AIRCRAFT_NUM VARCHAR2(7) CONSTRAINT RESERVATION_AIRCRAFT_NUM_FK REFERENCES AIRCRAFT(AIRCRAFT_NUM),
S_DAY VARCHAR2(8) NOT NULL,
S_TIME VARCHAR2(10),
E_TIME VARCHAR2(10),
DEPART VARCHAR2(15),
ARRIVE VARCHAR2(15));
